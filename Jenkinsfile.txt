pipeline {
    agent any
    tools {
        maven 'Maven3' // Use your configured Maven tool name
    }
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/Koushik0802/MyRepo.git'
            }
        }
        stage('Build with Maven') {
            steps {
                dir('simple-java') 
                    sh 'mvn clean package -DskipTests'
                }
            }
        }
        stage('Test') {
            steps {
                dir('simple-java') {
                    sh 'mvn test'
                }
            }
        }
    }
}
